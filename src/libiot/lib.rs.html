1//! # libiot - Rust IoT SDK
2//!
3//! A comprehensive Rust SDK that enables any IoT device to securely connect to the cloud,
4//! manage data, and interact with cloud services. This library is designed for embedded
5//! systems and supports `no_std` environments.
6//!
7//! ## Features
8//!
9//! ### Network Protocols
10//! - **HTTP Client**: RESTful API communication
11//! - **MQTT Client**: Lightweight publish-subscribe messaging
12//! - **MCP (Model Context Protocol)**: AI/LLM integration for embedded systems
13//! - **WebSocket**: Real-time bidirectional communication
14//! - **CoAP**: Constrained Application Protocol for IoT
15//!
16//! ### Storage Abstraction
17//! - Unified storage interface for various memory types
18//! - Support for EEPROM, Flash, SD cards, and RAM
19//! - Async and sync operation modes
20//!
21//! ### System Utilities
22//! - Command shell interface for embedded devices
23//! - Extensible command system with built-in help
24//!
25//! ## Usage
26//!
27//! Add this to your `Cargo.toml`:
28//!
29//! ```toml
30//! [dependencies]
31//! libiot = "0.1.0"
32//! ```
33//!
34//! ### Basic HTTP Client Example
35//!
36//! ```rust,no_run
37//! use libiot::network::application::http::{Client, Request, Method};
38//! # use libiot::network::Connection;
39//! # struct MockConnection;
40//! # impl Connection for MockConnection {}
41//! # impl libiot::network::Read for MockConnection {
42//! #     type Error = ();
43//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
44//! # }
45//! # impl libiot::network::Write for MockConnection {
46//! #     type Error = ();
47//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
48//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
49//! # }
50//! # impl libiot::network::Close for MockConnection {
51//! #     type Error = ();
52//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
53//! # }
54//!
55//! let connection = MockConnection;
56//! let mut client = Client::new(connection);
57//!
58//! let request = Request {
59//!     method: Method::Get,
60//!     path: "/api/data",
61//!     headers: heapless::Vec::new(),
62//!     body: None,
63//! };
64//!
65//! // let response = client.request(&request)?;
66//! ```
67//!
68//! ### MQTT Client Example
69//!
70//! ```rust,no_run
71//! use libiot::network::application::mqtt::{Client, Options, QoS};
72//! # use libiot::network::Connection;
73//! # struct MockConnection;
74//! # impl Connection for MockConnection {}
75//! # impl libiot::network::Read for MockConnection {
76//! #     type Error = ();
77//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
78//! # }
79//! # impl libiot::network::Write for MockConnection {
80//! #     type Error = ();
81//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
82//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
83//! # }
84//! # impl libiot::network::Close for MockConnection {
85//! #     type Error = ();
86//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
87//! # }
88//!
89//! let connection = MockConnection;
90//! let options = Options {
91//!     client_id: "my_device",
92//!     keep_alive_seconds: 60,
93//!     clean_session: true,
94//! };
95//!
96//! // let mut client = Client::connect(connection, options)?;
97//! // client.publish("sensors/temperature", b"23.5", QoS::AtMostOnce)?;
98//! ```
99//!
100//! ## Platform Support
101//!
102//! This library is designed to work on:
103//! - Embedded microcontrollers (ARM Cortex-M, RISC-V, etc.)
104//! - Linux-based IoT devices (Raspberry Pi, etc.)
105//! - Any platform supporting Rust's `core` library
106//!
107//! ## Optional Features
108//!
109//! - `std`: Enable standard library support (default: disabled)
110//! - `async`: Enable async/await support for non-blocking operations
111//! - `defmt`: Enable defmt logging support for embedded debugging
