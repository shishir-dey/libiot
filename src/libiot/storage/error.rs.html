8/// A common error type for storage operations.
9///
10/// This enum defines a set of common errors that can occur when working with
11/// storage devices. It is designed to be simple and portable for `no_std`
12/// environments while providing enough granularity for proper error handling
13/// across different storage technologies.
14///
15/// # Usage Examples
16///
17/// ```rust
18/// use libiot::storage::error::Error;
19///
20/// fn handle_storage_error(error: Error) {
21///     match error {
22///         Error::OutOfBounds => {
23///             println!("Attempted to access invalid address");
24///         }
25///         Error::WriteError => {
26///             println!("Failed to write data to storage");
27///         }
28///         Error::ReadError => {
29///             println!("Failed to read data from storage");
30///         }
31///         Error::EraseError => {
32///             println!("Failed to erase storage block");
33///         }
34///         Error::NotInitialized => {
35///             println!("Storage device not properly initialized");
36///         }
37///         Error::CardError => {
38///             println!("SD/MMC card specific error");
39///         }
40///         Error::StorageFault => {
41///             println!("Hardware fault detected in storage");
42///         }
43///     }
44/// }
45/// ```
46#[derive(Debug, PartialEq, Eq, Clone, Copy)]
47pub enum Error {
48    /// An operation was attempted on an address that is out of bounds.
49    ///
50    /// This error occurs when trying to access memory addresses beyond the
51    /// valid range of the storage device. It can happen with:
52    /// - Read/write operations beyond device capacity
53    /// - Invalid block or sector numbers
54    /// - Misaligned access patterns
55    OutOfBounds,
56
57    /// An error occurred during a write operation.
58    ///
59    /// Write errors can be caused by:
60    /// - Hardware failure during write
61    /// - Attempting to write to read-only storage
62    /// - Power loss during write operation
63    /// - Write protection enabled
64    /// - Storage device is full or worn out
65    WriteError,
66
67    /// An error occurred during a read operation.
68    ///
69    /// Read errors typically indicate:
70    /// - Data corruption due to aging or wear
71    /// - Hardware failure in the storage device
72    /// - Communication errors with the storage controller
73    /// - Power supply issues during read
74    ReadError,
75
76    /// An error occurred during an erase operation.
77    ///
78    /// Erase errors are common with flash memory and can indicate:
79    /// - Block is marked as bad and cannot be erased
80    /// - Hardware failure during erase cycle
81    /// - Erase verification failed
82    /// - Device reached end of life for erase cycles
83    EraseError,
84
85    /// An operation was attempted on a device that was not initialized.
86    ///
87    /// This error occurs when:
88    /// - Device initialization sequence was not completed
89    /// - Required setup operations were skipped
90    /// - Device was reset without proper re-initialization
91    /// - Communication with device could not be established
92    NotInitialized,
93
94    /// A card-specific error occurred (e.g., for SD/MMC cards).
95    ///
96    /// This error type covers SD and MMC card specific failures:
97    /// - Card detection/insertion issues
98    /// - Unsupported card type or format
99    /// - Card authentication or encryption errors
100    /// - Card command sequence errors
101    /// - Card protection switch activated
102    CardError,
103
104    /// The underlying storage is bad/unusable at a specific location.
105    ///
106    /// This indicates a hardware fault that makes the storage unreliable:
107    /// - Bad blocks in flash memory
108    /// - Corrupted sectors that cannot be recovered
109    /// - Physical damage to storage medium
110    /// - Excessive wear that makes area unusable
111    /// - Manufacturing defects discovered during operation
112    StorageFault,
113}

115#[cfg(feature = "defmt")]
116impl defmt::Format for Error {
117    fn format(&self, f: defmt::Formatter) {
118        match self {
119            Error::OutOfBounds => defmt::write!(f, "OutOfBounds"),
120            Error::WriteError => defmt::write!(f, "WriteError"),
121            Error::ReadError => defmt::write!(f, "ReadError"),
122            Error::EraseError => defmt::write!(f, "EraseError"),
123            Error::NotInitialized => defmt::write!(f, "NotInitialized"),
124            Error::CardError => defmt::write!(f, "CardError"),
125            Error::StorageFault => defmt::write!(f, "StorageFault"),
126        }
127    }
128}
