1//! # Network abstraction layer for embedded systems
2//!
3//! This module provides a comprehensive set of traits and implementations for working with
4//! different types of network devices in embedded systems. It includes traits for both
5//! synchronous and asynchronous operations, as well as support for various network protocols.
6//!
7//! ## Design Philosophy
8//!
9//! The network layer is designed around several core principles:
10//!
11//! - **Protocol Agnostic**: Core traits work with any underlying transport
12//! - **Zero-Cost Abstractions**: Traits compile down to direct function calls
13//! - **Embedded-First**: Designed for `no_std` environments with limited resources
14//! - **Composable**: Mix and match different connection types and protocols
15//!
16//! ## Architecture
17//!
18//! The network layer is organized into several abstraction levels:
19//!
20//! 1. **Core Traits** (`Read`, `Write`, `Close`, `Connection`)
21//! 2. **Connection Management** (`Connect`, `Bind`)
22//! 3. **Protocol-Specific Extensions** (`Http`, `Mqtt`, `WebSocket`, etc.)
23//! 4. **Application Layer** (protocol implementations)
24//!
25//! ## Usage Examples
26//!
27//! ### Basic Connection Usage
28//!
29//! ```rust,no_run
30//! use libiot::network::{Connection, Read, Write};
31//! # struct MockConnection;
32//! # impl Connection for MockConnection {}
33//! # impl libiot::network::Read for MockConnection {
34//! #     type Error = ();
35//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
36//! # }
37//! # impl libiot::network::Write for MockConnection {
38//! #     type Error = ();
39//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
40//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
41//! # }
42//! # impl libiot::network::Close for MockConnection {
43//! #     type Error = ();
44//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
45//! # }
46//!
47//! fn communicate_with_device<C: Connection>(mut conn: C) -> Result<(), C::Error> {
48//!     let data = b"Hello, device!";
49//!     conn.write(data)?;
50//!     conn.flush()?;
51//!     
52//!     let mut response = [0u8; 64];
53//!     let bytes_read = conn.read(&mut response)?;
54//!     
55//!     // Process response...
56//!     Ok(())
57//! }
58//! ```

84/// Trait for reading data from a network connection.
85///
86/// This trait provides a synchronous interface for reading data from any
87/// network connection. It's designed to be simple and efficient for embedded systems.
88///
89/// # Examples
90///
91/// ```rust,no_run
92/// use libiot::network::Read;
93/// # struct MockConnection;
94/// # impl Read for MockConnection {
95/// #     type Error = std::io::Error;
96/// #     fn read(&mut self, buf: &mut [u8]) -> Result<usize, Self::Error> {
97/// #         // Mock implementation
98/// #         Ok(0)
99/// #     }
100/// # }
101///
102/// fn read_data<R: Read>(reader: &mut R) -> Result<Vec<u8>, R::Error> {
103///     let mut buffer = [0u8; 1024];
104///     let bytes_read = reader.read(&mut buffer)?;
105///     Ok(buffer[..bytes_read].to_vec())
106/// }
107/// ```
108pub trait Read {
109    /// Associated error type for read operations
110    type Error: core::fmt::Debug;
111
112    /// Read data from the connection into the provided buffer.
113    ///
114    /// Returns the number of bytes read. A return value of 0 typically
115    /// indicates that the connection has been closed by the remote end.
116    ///
117    /// # Arguments
118    ///
119    /// * `buf` - Buffer to read data into
120    ///
121    /// # Returns
122    ///
123    /// * `Ok(n)` - Number of bytes read
124    /// * `Err(e)` - Read error occurred
125    fn read(&mut self, buf: &mut [u8]) -> Result<usize, Self::Error>;
126}

128/// Trait for writing data to a network connection.
129///
130/// This trait provides a synchronous interface for writing data to any
131/// network connection with support for flushing buffered data.
132///
133/// # Examples
134///
135/// ```rust,no_run
136/// use libiot::network::Write;
137/// # struct MockConnection;
138/// # impl Write for MockConnection {
139/// #     type Error = std::io::Error;
140/// #     fn write(&mut self, buf: &[u8]) -> Result<usize, Self::Error> {
141/// #         Ok(buf.len())
142/// #     }
143/// #     fn flush(&mut self) -> Result<(), Self::Error> {
144/// #         Ok(())
145/// #     }
146/// # }
147///
148/// fn send_message<W: Write>(writer: &mut W, message: &[u8]) -> Result<(), W::Error> {
149///     writer.write(message)?;
150///     writer.flush()?;
151///     Ok(())
152/// }
153/// ```
154pub trait Write {
155    /// Associated error type for write operations
156    type Error: core::fmt::Debug;
157
158    /// Write data to the connection.
159    ///
160    /// Returns the number of bytes written. The implementation may write
161    /// fewer bytes than requested.
162    ///
163    /// # Arguments
164    ///
165    /// * `buf` - Data to write
166    ///
167    /// # Returns
168    ///
169    /// * `Ok(n)` - Number of bytes written
170    /// * `Err(e)` - Write error occurred
171    fn write(&mut self, buf: &[u8]) -> Result<usize, Self::Error>;
172
173    /// Flush any buffered write data.
174    ///
175    /// This ensures that all buffered data is sent over the connection.
176    /// Some implementations may be no-ops if no buffering is used.
177    ///
178    /// # Returns
179    ///
180    /// * `Ok(())` - Flush completed successfully
181    /// * `Err(e)` - Flush error occurred
182    fn flush(&mut self) -> Result<(), Self::Error>;
183}

185/// Trait for closing network connections.
186///
187/// Provides a clean way to close connections and free associated resources.
188pub trait Close {
189    /// Associated error type for close operations
190    type Error: core::fmt::Debug;
191
192    /// Close the connection and free any associated resources.
193    ///
194    /// After calling this method, the connection should not be used for
195    /// further operations.
196    ///
197    /// # Returns
198    ///
199    /// * `Ok(())` - Connection closed successfully
200    /// * `Err(e)` - Error occurred while closing
201    fn close(self) -> Result<(), Self::Error>;
202}

204/// A complete synchronous network connection.
205///
206/// This trait combines reading, writing, and closing capabilities into a
207/// single unified interface. Any type implementing this trait can be used
208/// with protocol implementations.
209///
210/// # Examples
211///
212/// ```rust,no_run
213/// use libiot::network::Connection;
214///
215/// fn use_connection<C: Connection>(mut conn: C) {
216///     // Read, write, and eventually close the connection
217///     let mut buf = [0u8; 64];
218///     if let Ok(n) = conn.read(&mut buf) {
219///         // Process received data
220///     }
221///     
222///     if conn.write(b"response").is_ok() {
223///         let _ = conn.flush();
224///     }
225///     
226///     // Connection will be closed when dropped
227/// }
228/// ```
229pub trait Connection: Read + Write + Close {}

231/// Trait for establishing outbound network connections (client-side).
232///
233/// This trait is implemented by connection types that can establish
234/// connections to remote endpoints.
235pub trait Connect {
236    /// The type of connection that will be created
237    type Connection: Connection;
238    /// Associated error type for connection operations
239    type Error: core::fmt::Debug;
240
241    /// Establish a connection to a remote endpoint.
242    ///
243    /// # Arguments
244    ///
245    /// * `remote` - Address or identifier of the remote endpoint
246    ///
247    /// # Returns
248    ///
249    /// * `Ok(connection)` - Connection established successfully
250    /// * `Err(e)` - Failed to establish connection
251    fn connect(&mut self, remote: &str) -> Result<Self::Connection, Self::Error>;
252}
253
254/// Trait for accepting inbound network connections (server-side).
255///
256/// This trait is implemented by connection types that can listen for
257/// and accept incoming connections.
258pub trait Bind {
259    /// The type of connection that will be created for each client
260    type Connection: Connection;
261    /// Associated error type for bind operations
262    type Error: core::fmt::Debug;
263
264    /// Bind to a local address and accept incoming connections.
265    ///
266    /// This is typically a blocking operation that waits for a client
267    /// to connect.
268    ///
269    /// # Arguments
270    ///
271    /// * `local` - Local address to bind to
272    ///
273    /// # Returns
274    ///
275    /// * `Ok(connection)` - Incoming connection accepted
276    /// * `Err(e)` - Failed to bind or accept connection
277    fn bind(&mut self, local: &str) -> Result<Self::Connection, Self::Error>;
278}

284/// Trait for reading data from a network connection asynchronously.
285///
286/// This is the async equivalent of the [`Read`] trait, designed for
287/// non-blocking I/O operations in async contexts.
288#[cfg(feature = "async")]

290    /// Associated error type for async read operations
291    type Error: core::fmt::Debug;
292
293    /// Read data from the connection asynchronously.

299    /// # Returns

304}

306/// Trait for writing data to a network connection asynchronously.

309/// non-blocking I/O operations in async contexts.

413/// Trait for UDP socket operations.
414///
415/// UDP is a connectionless protocol, so it uses a different interface
416/// than connection-based protocols.
417pub trait UdpSocket {
418    /// Associated error type for UDP operations
419    type Error: core::fmt::Debug;
420
421    /// Send data to a specific remote endpoint.
422    ///
423    /// # Arguments
424    ///
425    /// * `remote` - Address of the remote endpoint
426    /// * `buf` - Data to send
427    ///
428    /// # Returns
429    ///
430    /// * `Ok(n)` - Number of bytes sent
431    /// * `Err(e)` - Send error occurred
432    fn send_to(&mut self, remote: &str, buf: &[u8]) -> Result<usize, Self::Error>;
433
434    /// Receive data from any remote endpoint.
435    ///
436    /// # Arguments
437    ///
438    /// * `buf` - Buffer to receive data into
439    ///
440    /// # Returns
441    ///
442    /// * `Ok((n, addr))` - Number of bytes received and sender address
443    /// * `Err(e)` - Receive error occurred
444    fn recv_from(&mut self, buf: &mut [u8]) -> Result<(usize, &str), Self::Error>;
445}
446
447/// Trait for asynchronous UDP socket operations.
448#[cfg(feature = "async")]
449pub trait AsyncUdpSocket {
450    /// Associated error type for async UDP operations
451    type Error: core::fmt::Debug;
452
453    /// Send data to a specific remote endpoint asynchronously.
454    async fn send_to(&mut self, remote: &str, buf: &[u8]) -> Result<usize, Self::Error>;
455
456    /// Receive data from any remote endpoint asynchronously.
457    async fn recv_from(&mut self, buf: &mut [u8]) -> Result<(usize, &str), Self::Error>;
458}
