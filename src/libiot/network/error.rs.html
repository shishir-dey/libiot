6/// A common error type for network operations.
7///
8/// This enum defines a set of common errors that can occur when working with
9/// network devices. It is designed to be simple and portable for `no_std`
10/// environments while providing enough detail for proper error handling.
11///
12/// # Examples
13///
14/// ```rust
15/// use libiot::network::error::Error;
16///
17/// fn handle_network_error(error: Error) {
18///     match error {
19///         Error::ConnectionRefused => {
20///             println!("Connection was refused by the remote server");
21///         }
22///         Error::Timeout => {
23///             println!("Operation timed out");
24///         }
25///         Error::WriteError => {
26///             println!("Failed to write data to the connection");
27///         }
28///         _ => {
29///             println!("Other network error occurred");
30///         }
31///     }
32/// }
33/// ```
34#[derive(Debug, PartialEq, Eq, Clone, Copy)]
35pub enum Error {
36    /// An operation was attempted on a connection that is not open.
37    ///
38    /// This error occurs when trying to read from, write to, or perform other
39    /// operations on a connection that hasn't been established or has been closed.
40    NotOpen,
41
42    /// An error occurred during a write operation.
43    ///
44    /// This can happen due to network issues, buffer overflow, or the remote
45    /// end closing the connection during a write operation.
46    WriteError,
47
48    /// An error occurred during a read operation.
49    ///
50    /// This can happen due to network issues, malformed data, or the remote
51    /// end closing the connection during a read operation.
52    ReadError,
53
54    /// A connection attempt was refused by the remote server.
55    ///
56    /// This typically occurs when the remote server is not accepting connections,
57    /// the port is closed, or authentication fails.
58    ConnectionRefused,
59
60    /// A timeout occurred during a network operation.
61    ///
62    /// This happens when an operation takes longer than the configured timeout
63    /// period, which could indicate network congestion or unresponsive remote endpoints.
64    Timeout,
65
66    /// The connection was closed unexpectedly.
67    ///
68    /// This occurs when the remote endpoint closes the connection, either gracefully
69    /// or due to an error condition.
70    ConnectionClosed,
71
72    /// An invalid address was provided.
73    ///
74    /// This error is returned when the provided network address is malformed,
75    /// unreachable, or otherwise invalid.
76    InvalidAddress,
77
78    /// A protocol-specific error occurred.
79    ///
80    /// This is a catch-all error for protocol-specific issues such as:
81    /// - Malformed protocol messages
82    /// - Unsupported protocol versions
83    /// - Protocol state violations
84    /// - Invalid protocol parameters
85    ProtocolError,
86}

88#[cfg(feature = "defmt")]
89impl defmt::Format for Error {
90    fn format(&self, f: defmt::Formatter) {
91        match self {
92            Error::NotOpen => defmt::write!(f, "NotOpen"),
93            Error::WriteError => defmt::write!(f, "WriteError"),
94            Error::ReadError => defmt::write!(f, "ReadError"),
95            Error::ConnectionRefused => defmt::write!(f, "ConnectionRefused"),
96            Error::Timeout => defmt::write!(f, "Timeout"),
97            Error::ConnectionClosed => defmt::write!(f, "ConnectionClosed"),
98            Error::InvalidAddress => defmt::write!(f, "InvalidAddress"),
99            Error::ProtocolError => defmt::write!(f, "ProtocolError"),
100        }
101    }
102}
