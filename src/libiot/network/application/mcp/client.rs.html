1//! MCP Client implementation for embedded systems
2
3use super::*;
4use crate::network::{Connection, error::Error as NetworkError};
5use heapless::Vec;
6
7/// MCP Client that works over any connection type
8pub struct McpClient<C, H>
9where
10    C: Connection,
11    H: McpHandler,
12{
13    connection: C,
14    registry: FunctionRegistry<H>,
15    buffer: Vec<u8, 1024>,
16}
17
18impl<C, H> McpClient<C, H>
19where
20    C: Connection,
21    H: McpHandler,
22{
23    /// Create a new MCP client with a connection and function registry
24    pub fn new(connection: C, registry: FunctionRegistry<H>) -> Self {
25        Self {
26            connection,
27            registry,
28            buffer: Vec::new(),
29        }
30    }
31
32    /// Process incoming MCP messages and return responses
33    pub fn process_message(&mut self) -> Result<(), NetworkError> {
34        // Clear buffer for new message
35        self.buffer.clear();
36
37        // Read incoming data
38        let mut temp_buf = [0u8; 256];
39        loop {
40            match self.connection.read(&mut temp_buf) {
41                Ok(0) => break, // No more data
42                Ok(n) => {
43                    if self.buffer.extend_from_slice(&temp_buf[..n]).is_err() {
44                        return Err(NetworkError::ReadError);
45                    }
46
47                    // Check if we have a complete JSON message
48                    if self.has_complete_message() {
49                        break;
50                    }
51                }
52                Err(_) => return Err(NetworkError::ReadError),
53            }
54        }
55
56        if self.buffer.is_empty() {
57            return Ok(());
58        }
59
60        // Parse and handle the message
61        let response = self.handle_message();
62
63        // Send response back
64        self.send_response(&response)
65    }
66
67    /// Check if buffer contains a complete JSON message
68    fn has_complete_message(&self) -> bool {
69        let mut brace_count = 0;
70        let mut in_string = false;
71        let mut escape_next = false;
72
73        for &byte in &self.buffer {
74            if escape_next {
75                escape_next = false;
76                continue;
77            }
78
79            match byte {
80                b'\\' if in_string => escape_next = true,
81                b'"' => in_string = !in_string,
82                b'{' if !in_string => brace_count += 1,
83                b'}' if !in_string => {
84                    if brace_count > 0 {
85                        brace_count -= 1;
86                        if brace_count == 0 {
87                            return true;
88                        }
89                    }
90                    // If brace_count is 0 and we encounter '}', ignore it
91                    // as it indicates malformed JSON (extra closing brace)
92                }
93                _ => {}
94            }
95        }
96
97        false
98    }
99
100    /// Parse and handle an MCP message
101    fn handle_message(&mut self) -> McpResponse {
102        // Try to parse the JSON message
103        let message_str = match core::str::from_utf8(&self.buffer) {
104            Ok(s) => s,
105            Err(_) => {
106                return McpResponse {
107                    status: ResponseStatus::Error,
108                    error: Some(heapless::String::try_from("Invalid UTF-8").unwrap_or_default()),
109                    result: None,
110                };
111            }
112        };
113
114        // Parse the MCP message
115        match serde_json_core::from_str::<McpMessage>(message_str) {
116            Ok((message, _)) => {
117                // Execute the function
118                self.registry.execute(message.function, message.arguments)
119            }
120            Err(_) => McpResponse {
121                status: ResponseStatus::Error,
122                error: Some(heapless::String::try_from("JSON parse error").unwrap_or_default()),
123                result: None,
124            },
125        }
126    }
127
128    /// Send response back over the connection
129    fn send_response(&mut self, response: &McpResponse) -> Result<(), NetworkError> {
130        // Serialize response to JSON
131        let mut response_buf = [0u8; 512];
132        match serde_json_core::to_slice(response, &mut response_buf) {
133            Ok(len) => {
134                // Send the response
135                self.connection
136                    .write(&response_buf[..len])
137                    .map_err(|_| NetworkError::WriteError)?;
138                self.connection
139                    .flush()
140                    .map_err(|_| NetworkError::WriteError)?;
141                Ok(())
142            }
143            Err(_) => Err(NetworkError::WriteError),
144        }
145    }
146
147    /// Get a mutable reference to the function registry
148    pub fn registry_mut(&mut self) -> &mut FunctionRegistry<H> {
149        &mut self.registry
150    }
151
152    /// Get the underlying connection
153    pub fn connection(&self) -> &C {
154        &self.connection
155    }
156
157    /// Get a mutable reference to the underlying connection
158    pub fn connection_mut(&mut self) -> &mut C {
159        &mut self.connection
160    }
161}
