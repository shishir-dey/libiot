1//! HTTP/1.1 client implementation for embedded systems.
2//!
3//! This module provides a lightweight HTTP client designed for `no_std` environments.
4//! It supports basic HTTP operations with fixed-size buffers and minimal memory usage.
5//!
6//! # Features
7//!
8//! - HTTP/1.1 protocol support
9//! - GET and POST methods
10//! - Custom headers
11//! - Request/response body handling
12//! - Connection reuse
13//! - Fixed-size buffers for predictable memory usage
14//!
15//! # Limitations
16//!
17//! - Only supports HTTP/1.1 (no HTTP/2 or HTTP/3)
18//! - Limited to GET and POST methods
19//! - Maximum header count and sizes are compile-time constants
20//! - Response body size is limited by buffer capacity
21//! - No automatic redirect handling
22//! - No persistent connection management
23//!
24//! # Examples
25//!
26//! ## Simple GET Request
27//!
28//! ```rust,no_run
29//! use libiot::network::application::http::{Client, Request, Method};
30//! # use libiot::network::Connection;
31//! # struct MockConnection;
32//! # impl Connection for MockConnection {}
33//! # impl libiot::network::Read for MockConnection {
34//! #     type Error = ();
35//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
36//! # }
37//! # impl libiot::network::Write for MockConnection {
38//! #     type Error = ();
39//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
40//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
41//! # }
42//! # impl libiot::network::Close for MockConnection {
43//! #     type Error = ();
44//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
45//! # }
46//!
47//! let connection = MockConnection;
48//! let mut client = Client::new(connection);
49//!
50//! let request = Request {
51//!     method: Method::Get,
52//!     path: "/api/sensors",
53//!     headers: heapless::Vec::new(),
54//!     body: None,
55//! };
56//!
57//! // let response = client.request(&request)?;
58//! // println!("Status: {}", response.status_code);
59//! ```
60//!
61//! ## POST Request with JSON Body
62//!
63//! ```rust,no_run
64//! use libiot::network::application::http::{Client, Request, Method, Header};
65//! # use libiot::network::Connection;
66//! # struct MockConnection;
67//! # impl Connection for MockConnection {}
68//! # impl libiot::network::Read for MockConnection {
69//! #     type Error = ();
70//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
71//! # }
72//! # impl libiot::network::Write for MockConnection {
73//! #     type Error = ();
74//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
75//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
76//! # }
77//! # impl libiot::network::Close for MockConnection {
78//! #     type Error = ();
79//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
80//! # }
81//!
82//! let connection = MockConnection;
83//! let mut client = Client::new(connection);
84//!
85//! let json_data = br#"{"temperature": 23.5, "humidity": 65}"#;
86//! let mut headers = heapless::Vec::new();
87//!
88//! let content_type_header = Header {
89//!     name: heapless::String::try_from("Content-Type").unwrap(),
90//!     value: heapless::String::try_from("application/json").unwrap(),
91//! };
92//! headers.push(content_type_header).unwrap();
93//!
94//! let request = Request {
95//!     method: Method::Post,
96//!     path: "/api/data",
97//!     headers,
98//!     body: Some(json_data),
99//! };
100//!
101//! // let response = client.request(&request)?;
102//! ```

109/// Maximum number of headers allowed per request/response.
110const MAX_HEADERS: usize = 16;
111
112/// Maximum length for header names in bytes.
113const MAX_HEADER_NAME_LEN: usize = 64;
114
115/// Maximum length for header values in bytes.
116const MAX_HEADER_VALUE_LEN: usize = 256;
117
118/// HTTP request methods supported by the client.
119///
120/// Currently supports the most common HTTP methods used in IoT applications.
121/// Additional methods can be added as needed.
122///
123/// # Examples
124///
125/// ```rust
126/// use libiot::network::application::http::Method;
127///
128/// let get_method = Method::Get;
129/// let post_method = Method::Post;
130///
131/// assert_eq!(get_method.as_str(), "GET");
132/// assert_eq!(post_method.as_str(), "POST");
133/// ```
134#[derive(Debug, Clone, Copy, PartialEq, Eq)]
135pub enum Method {
136    /// HTTP GET method for retrieving data.
137    Get,
138    /// HTTP POST method for sending data.
139    Post,
140}
141
142impl Method {
143    /// Convert the method to its string representation.
144    ///
145    /// Returns the standard HTTP method name as used in request lines.
146    ///
147    /// # Examples
148    ///
149    /// ```rust
150    /// use libiot::network::application::http::Method;
151    ///
152    /// assert_eq!(Method::Get.as_str(), "GET");
153    /// assert_eq!(Method::Post.as_str(), "POST");
154    /// ```
155    pub fn as_str(&self) -> &'static str {
156        match self {
157            Method::Get => "GET",
158            Method::Post => "POST",
159        }
160    }
161}

163/// An HTTP header consisting of a name-value pair.
164///
165/// Headers are used to pass additional information with HTTP requests and responses.
166/// Both the name and value are stored as heap-free strings with compile-time size limits.
167///
168/// # Examples
169///
170/// ```rust
171/// use libiot::network::application::http::Header;
172/// use heapless::String;
173///
174/// let header = Header {
175///     name: String::try_from("Content-Type").unwrap(),
176///     value: String::try_from("application/json").unwrap(),
177/// };
178///
179/// assert_eq!(header.name.as_str(), "Content-Type");
180/// assert_eq!(header.value.as_str(), "application/json");
181/// ```
182#[derive(Debug, Clone)]
183pub struct Header {
184    /// The header name (e.g., "Content-Type", "Authorization").
185    pub name: String<MAX_HEADER_NAME_LEN>,
186    /// The header value (e.g., "application/json", "Bearer token123").
187    pub value: String<MAX_HEADER_VALUE_LEN>,
188}
189
190/// An HTTP request to be sent by the client.
191///
192/// Contains all the information needed to construct a complete HTTP request,
193/// including method, path, headers, and optional body data.
194///
195/// # Examples
196///
197/// ```rust
198/// use libiot::network::application::http::{Request, Method};
199/// use heapless::Vec;
200///
201/// let request = Request {
202///     method: Method::Get,
203///     path: "/api/status",
204///     headers: Vec::new(),
205///     body: None,
206/// };
207/// ```
208pub struct Request<'a> {
209    /// The HTTP method to use for this request.
210    pub method: Method,
211    /// The request path (e.g., "/api/data", "/status").
212    pub path: &'a str,
213    /// Optional headers to include with the request.
214    pub headers: Vec<Header, MAX_HEADERS>,
215    /// Optional request body data.
216    pub body: Option<&'a [u8]>,
217}
218
219/// An HTTP response received from the server.
220///
221/// Contains the response status code, headers, and body data returned by the server.
222/// The body size is limited by the internal buffer capacity.
223///
224/// # Examples
225///
226/// ```rust
227/// use libiot::network::application::http::Response;
228///
229/// // Response is typically created by the HTTP client
230/// // let response = client.request(&request)?;
231/// //
232/// // if response.status_code == 200 {
233/// //     println!("Success! Body length: {}", response.body.len());
234/// // }
235/// ```
236#[derive(Debug)]
237pub struct Response {
238    /// HTTP status code (e.g., 200, 404, 500).
239    pub status_code: u16,
240    /// Response headers sent by the server.
241    pub headers: Vec<Header, MAX_HEADERS>,
242    /// Response body data with a maximum size of 2048 bytes.
243    pub body: Vec<u8, 2048>,
244}

246/// HTTP client for making requests over any connection type.
247///
248/// The client is generic over the connection type, allowing it to work with
249/// different transport layers (TCP, TLS, etc.) as long as they implement
250/// the [`Connection`] trait.
251///
252/// # Type Parameters
253///
254/// * `C` - The connection type implementing [`Connection`]
255///
256/// # Examples
257///
258/// ```rust,no_run
259/// use libiot::network::application::http::Client;
260/// # use libiot::network::Connection;
261/// # struct MockConnection;
262/// # impl Connection for MockConnection {}
263/// # impl libiot::network::Read for MockConnection {
264/// #     type Error = ();
265/// #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
266/// # }
267/// # impl libiot::network::Write for MockConnection {
268/// #     type Error = ();
269/// #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
270/// #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
271/// # }
272/// # impl libiot::network::Close for MockConnection {
273/// #     type Error = ();
274/// #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
275/// # }
276///
277/// let connection = MockConnection;
278/// let client = Client::new(connection);
279/// // Now ready to make HTTP requests
280/// ```
281pub struct Client<C: Connection> {
282    connection: C,
283}

329    /// Send an HTTP request and receive the response.
330    ///
331    /// This method constructs a complete HTTP request from the provided [`Request`],
332    /// sends it over the connection, and parses the response. The operation is
333    /// synchronous and will block until the response is received or an error occurs.
334    ///
335    /// # Arguments
336    ///
337    /// * `request` - The HTTP request to send
338    ///
339    /// # Returns
340    ///
341    /// * `Ok(response)` - The HTTP response from the server
342    /// * `Err(error)` - Network or protocol error occurred
343    ///
344    /// # Errors
345    ///
346    /// This method can return various errors:
347    ///
348    /// * [`Error::WriteError`] - Failed to send the request
349    /// * [`Error::ReadError`] - Failed to read the response
350    /// * [`Error::ConnectionClosed`] - Connection was closed unexpectedly
351    /// * [`Error::ProtocolError`] - Invalid HTTP response format
352    ///
353    /// # Examples
354    ///
355    /// ```rust,no_run
356    /// use libiot::network::application::http::{Client, Request, Method};
357    /// # use libiot::network::Connection;
358    /// # struct MockConnection;
359    /// # impl Connection for MockConnection {}
360    /// # impl libiot::network::Read for MockConnection {
361    /// #     type Error = ();
362    /// #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
363    /// # }
364    /// # impl libiot::network::Write for MockConnection {
365    /// #     type Error = ();
366    /// #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
367    /// #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
368    /// # }
369    /// # impl libiot::network::Close for MockConnection {
370    /// #     type Error = ();
371    /// #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
372    /// # }
373    ///
374    /// let connection = MockConnection;
375    /// let mut client = Client::new(connection);
376    ///
377    /// let request = Request {
378    ///     method: Method::Get,
379    ///     path: "/api/temperature",
380    ///     headers: heapless::Vec::new(),
381    ///     body: None,
382    /// };
383    ///
384    /// // match client.request(&request) {
385    /// //     Ok(response) => {
386    /// //         if response.status_code == 200 {
387    /// //             println!("Temperature data: {:?}", response.body);
388    /// //         }
389    /// //     }
390    /// //     Err(e) => println!("Request failed: {:?}", e),
391    /// // }
392    /// ```
393    pub fn request(&mut self, request: &Request) -> Result<Response, Error> {
394        // --- Build Request ---
395        let mut request_buf: Vec<u8, 2048> = Vec::new();
396
397        // Request line
398        request_buf
399            .extend_from_slice(request.method.as_str().as_bytes())
400            .map_err(|_| Error::WriteError)?;
401        request_buf.push(b' ').map_err(|_| Error::WriteError)?;
402        request_buf
403            .extend_from_slice(request.path.as_bytes())
404            .map_err(|_| Error::WriteError)?;
405        request_buf
406            .extend_from_slice(b" HTTP/1.1\r\n")
407            .map_err(|_| Error::WriteError)?;
408
409        // Headers
410        let mut has_user_agent = false;
411        for header in &request.headers {
412            if header.name.eq_ignore_ascii_case("User-Agent") {
413                has_user_agent = true;
414            }
415            request_buf
416                .extend_from_slice(header.name.as_bytes())
417                .map_err(|_| Error::WriteError)?;
418            request_buf
419                .extend_from_slice(b": ")
420                .map_err(|_| Error::WriteError)?;
421            request_buf
422                .extend_from_slice(header.value.as_bytes())
423                .map_err(|_| Error::WriteError)?;
424            request_buf
425                .extend_from_slice(b"\r\n")
426                .map_err(|_| Error::WriteError)?;
427        }
428
429        if !has_user_agent {
430            request_buf
431                .extend_from_slice(b"User-Agent:;\r\n")
432                .map_err(|_| Error::WriteError)?;
433        }
434
435        // Body
436        if let Some(body) = request.body {
437            let mut len_str: String<10> = String::new();
438            write!(len_str, "{}", body.len()).unwrap();
439
440            request_buf
441                .extend_from_slice(b"Content-Length: ")
442                .map_err(|_| Error::WriteError)?;
443            request_buf
444                .extend_from_slice(len_str.as_bytes())
445                .map_err(|_| Error::WriteError)?;
446            request_buf
447                .extend_from_slice(b"\r\n\r\n")
448                .map_err(|_| Error::WriteError)?;
449            request_buf
450                .extend_from_slice(body)
451                .map_err(|_| Error::WriteError)?;
452        } else {
453            request_buf
454                .extend_from_slice(b"\r\n")
455                .map_err(|_| Error::WriteError)?;
456        }
457
458        // --- Send Request ---
459        self.connection
460            .write(&request_buf)
461            .map_err(|_| Error::WriteError)?;
462        self.connection.flush().map_err(|_| Error::WriteError)?;
463
464        // --- Receive Response ---
465        let mut response_buf = [0u8; 2048];
466        let mut total_read = 0;
467        loop {
468            match self.connection.read(&mut response_buf[total_read..]) {
469                Ok(0) if total_read > 0 => break, // Connection closed, but we have data
470                Ok(0) => return Err(Error::ConnectionClosed),
471                Ok(n) => {
472                    total_read += n;
473                    if total_read >= response_buf.len() {
474                        break;
475                    }
476                    // This is a simplistic check. A robust client would parse Content-Length
477                    // and continue reading until the body is fully received.
478                    if find_slice(&response_buf[..total_read], b"\r\n\r\n").is_some() {
479                        // For now, we assume the first read gets all headers and maybe start of body
480                        break;
481                    }
482                }
483                Err(_) => return Err(Error::ReadError),
484            }
485        }
486
487        // --- Parse Response ---
488        let response_data = &response_buf[..total_read];
489
490        // Find where headers end and body begins
491        let header_end_pos = find_slice(response_data, b"\r\n\r\n").ok_or(Error::ProtocolError)?;
492        let header_data = &response_data[..header_end_pos];
493        let body_data = &response_data[header_end_pos + 4..];
494
495        let header_str = core::str::from_utf8(header_data).map_err(|_| Error::ProtocolError)?;
496        let mut lines = header_str.lines();
497
498        // Parse status line
499        let status_line = lines.next().ok_or(Error::ProtocolError)?;
500        let mut status_parts = status_line.splitn(3, ' ');
501        status_parts.next(); // Skip HTTP version
502        let status_code_str = status_parts.next().ok_or(Error::ProtocolError)?;
503        let status_code = status_code_str
504            .parse::<u16>()
505            .map_err(|_| Error::ProtocolError)?;
506
507        // Parse headers
508        let mut response_headers: Vec<Header, MAX_HEADERS> = Vec::new();
509        let mut content_length: Option<usize> = None;
510
511        for line in lines {
512            if line.is_empty() {
513                continue;
514            }
515            let mut parts = line.splitn(2, ':');
516            let name = parts.next().ok_or(Error::ProtocolError)?.trim();
517            let value = parts.next().ok_or(Error::ProtocolError)?.trim();
518
519            if name.eq_ignore_ascii_case("Content-Length") {
520                content_length = value.parse::<usize>().ok();
521            }
522
523            response_headers
524                .push(Header {
525                    name: String::try_from(name).map_err(|_| Error::ProtocolError)?,
526                    value: String::try_from(value).map_err(|_| Error::ProtocolError)?,
527                })
528                .map_err(|_| Error::ProtocolError)?;
529        }
530
531        let mut body = Vec::from_slice(body_data).map_err(|_| Error::ProtocolError)?;
532        if let Some(len) = content_length {
533            while body.len() < len {
534                if body.len() == body.capacity() {
535                    // Body is larger than our buffer.
536                    return Err(Error::ProtocolError);
537                }
538
539                // Read more data into a temporary buffer, then extend our body vec.
540                let mut temp_buf = [0; 256];
541                let remaining_len = len - body.len();
542                let read_len = core::cmp::min(remaining_len, temp_buf.len());
543                if read_len == 0 {
544                    break;
545                }
546
547                match self.connection.read(&mut temp_buf[..read_len]) {
548                    Ok(0) => return Err(Error::ConnectionClosed), // Prematurely closed
549                    Ok(n) => {
550                        if body.extend_from_slice(&temp_buf[..n]).is_err() {
551                            return Err(Error::ProtocolError); // Should not happen given capacity check
552                        }
553                    }
554                    Err(_) => return Err(Error::ReadError),
555                }
556            }
557
558            // Truncate to ensure we have exactly `len` bytes.
559            if body.len() > len {
560                body.truncate(len);
561            }
562        }
563
564        Ok(Response {
565            status_code,
566            headers: response_headers,
567            body,
568        })
569    }
570}

572/// Find the first occurrence of a slice in another slice and return its starting position.
573///
574/// This is a utility function used internally for parsing HTTP responses to locate
575/// the boundary between headers and body.
576///
577/// # Arguments
578///
579/// * `haystack` - The slice to search in
580/// * `needle` - The slice to search for
581///
582/// # Returns
583///
584/// * `Some(index)` - The starting position of the first occurrence
585/// * `None` - The needle was not found in the haystack
586fn find_slice(haystack: &[u8], needle: &[u8]) -> Option<usize> {
587    haystack
588        .windows(needle.len())
589        .position(|window| window == needle)
590}
