1//! HTTP/1.1 protocol implementation for embedded systems.
2//!
3//! This module provides a lightweight HTTP client implementation designed specifically
4//! for embedded systems and `no_std` environments. It focuses on simplicity,
5//! predictable memory usage, and compatibility with resource-constrained devices.
6//!
7//! # Features
8//!
9//! - HTTP/1.1 protocol compliance
10//! - Synchronous request/response model
11//! - Fixed-size buffers for predictable memory usage
12//! - Support for custom headers
13//! - GET and POST method support
14//! - Connection reuse capability
15//!
16//! # Usage
17//!
18//! The main entry point is the [`client::Client`] which works with any connection
19//! type implementing the [`crate::network::Connection`] trait.
20//!
21//! ```rust,no_run
22//! use libiot::network::application::http::{Client, Request, Method};
23//! # use libiot::network::Connection;
24//! # struct MockConnection;
25//! # impl Connection for MockConnection {}
26//! # impl libiot::network::Read for MockConnection {
27//! #     type Error = ();
28//! #     fn read(&mut self, _buf: &mut [u8]) -> Result<usize, Self::Error> { Ok(0) }
29//! # }
30//! # impl libiot::network::Write for MockConnection {
31//! #     type Error = ();
32//! #     fn write(&mut self, _buf: &[u8]) -> Result<usize, Self::Error> { Ok(0) }
33//! #     fn flush(&mut self) -> Result<(), Self::Error> { Ok(()) }
34//! # }
35//! # impl libiot::network::Close for MockConnection {
36//! #     type Error = ();
37//! #     fn close(self) -> Result<(), Self::Error> { Ok(()) }
38//! # }
39//!
40//! let connection = MockConnection;
41//! let mut client = Client::new(connection);
42//!
43//! let request = Request {
44//!     method: Method::Get,
45//!     path: "/api/status",
46//!     headers: heapless::Vec::new(),
47//!     body: None,
48//! };
49//!
50//! // let response = client.request(&request)?;
51//! ```
